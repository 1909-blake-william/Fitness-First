/************************************
Tables and sequences
************************************/

DROP TABLE Points;
CREATE TABLE Points (
    points_id INT PRIMARY KEY,
    actual_points INT
);
DROP TABLE Muscle;
CREATE TABLE Muscle (
    muscle_id INT PRIMARY KEY,
    muscle_name VARCHAR2(30)
);
DROP TABLE Exercise_Type;
CREATE TABLE Exercise_Type (
    exercise_type_id INT PRIMARY KEY,
    exercis_type_name VARCHAR2 (40)
);
DROP TABLE User_level;
CREATE TABLE User_level (
    level_id INT PRIMARY KEY,
    level_identifier INT
);
DROP TABLE Difficulty;
CREATE TABLE Difficulty (
    difficulty_id INT PRIMARY KEY,
    difficulty_name VARCHAR2 (20),
    difficulty_points NUMBER REFERENCES Points(points_id)
);
DROP SEQUENCE Exercise_id_seq;
DROP TABLE Exercise;
CREATE SEQUENCE Exercise_id_seq;
CREATE TABLE Exercise (
    exercise_id INT PRIMARY KEY,
    exercise_name VARCHAR2(20) ,
    muscle VARCHAR(40),
    exercise_type INT REFERENCES Exercise_Type(exercise_type_id),
    exercise_description VARCHAR2(250),
    exercise_difficulty INT REFERENCES Difficulty(difficulty_id)
);
DROP SEQUENCE WGER_User_id_seq;
DROP TABLE WGER_User;
CREATE SEQUENCE WGER_User_id_seq;
CREATE TABLE WGER_User (
    user_id INT PRIMARY KEY,
    username VARCHAR2 (20) UNIQUE NOT NULL,
    user_password VARCHAR(50) NOT NULL,
    firstname VARCHAR(20),
    lastname VARCHAR(20),
    email VARCHAR(30),
    gender VARCHAR2(5),
    height INT,
    weight INT,
    considerations VARCHAR2(250),
    user_level INT REFERENCES User_level(level_id),
    points INT REFERENCES Points(points_id)
);
DROP SEQUENCE Workout_id_seq;
DROP TABLE Workout;
CREATE SEQUENCE Workout_id_seq;  
CREATE TABLE Workout (
    workout_id INT PRIMARY KEY,
    exercises INT REFERENCES Exercise(exercise_id),
    exercise_sets INT,
    repetitions INT,
    workout_user INT REFERENCES Wger_User(user_id),
    uses_weights NUMBER(1) DEFAULT 0 NOT NULL,
    completed NUMBER(1) DEFAULT 0 NOT NULL
);
